通过sleep（1）方法实现，连续10秒钟空闲，则停止

1、void Startserver()

{
 SendAndGetStatus = true;
   while(SendAndGetStatus){
      GetAndSend();
}
}

2、void GetAndSend(){
int nRet;
 if ( nRet == false )
{
  Sleep(1);
   StopNum++;
  //连续10秒没有接收到消息，结束循环
   if(StopNum>=10000)
    SendAndGetStatus=false;
   return ;
}

if(nRet==true){
   StopNum=0;//赋值为0

   }
}
